在网页中使用Audio API - 开始使用

狐妖i 前端Web Audio API

话不多说直接上代码

代码示例

HTML

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <title></title>
    </head>
    <body>
        <audio controls="true" src="忘川风华录,海伊,诗岸 - 青鸟衔风.mp3"></audio>
    </body>
</html>

Javascipt

//创建API
var audioCtx = new(window.AudioContext || window.webkitAudioContext)();
//标签链接至API (audio标签或是video标签)
var source = audioCtx.createMediaElementSource(document.querySelector("audio"))
//链接到输出
source.connect(audioCtx.destination);
